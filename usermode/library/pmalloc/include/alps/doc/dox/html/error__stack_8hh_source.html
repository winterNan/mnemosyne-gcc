278 inline ErrorStack::ErrorStack(const ErrorStack &other, const char* filename,
279  const char* func, uint32_t linenum, const char* more_custom_message)
280  : custom_message_(CXX11_NULLPTR) {
281  // Invariant: if kErrorCodeOk, no more processing
282  if (LIKELY(other.error_code_ == kErrorCodeOk)) {
283  this->error_code_ = kErrorCodeOk;
284  return;
285  }
286 
287  operator=(other);
288  // augment stacktrace
289  if (stack_depth_ != 0 && stack_depth_ < kMaxStackDepth) {
290  filenames_[stack_depth_] = filename;
291  funcs_[stack_depth_] = func;
292  linenums_[stack_depth_] = linenum;
293  ++stack_depth_;
294  }
295  // augment custom error message
296  if (more_custom_message) {
297  append_custom_message(more_custom_message);
298  }
299 }

301 inline ErrorStack& ErrorStack::operator=(const ErrorStack &other) {
302  // Invariant: if kErrorCodeOk, no more processing
303  if (LIKELY(other.error_code_ == kErrorCodeOk)) {
304  this->error_code_ = kErrorCodeOk;
305  return *this;
306  }
307 
308  // this copy assignment is actually a move assignment.
309  // checked_/custom_message_ are mutable for that.
310  custom_message_ = other.custom_message_; // steal.
311  other.custom_message_ = CXX11_NULLPTR; // simply stolen. much more efficient.
312  stack_depth_ = other.stack_depth_;
313  for (int i = 0; i < other.stack_depth_; ++i) {
314  filenames_[i] = other.filenames_[i];
315  funcs_[i] = other.funcs_[i];
316  linenums_[i] = other.linenums_[i];
317  }
318  os_errno_ = other.os_errno_;
319  error_code_ = other.error_code_;
320  checked_ = false;
321  other.checked_ = true;
322  return *this;
323 }

345 inline void ErrorStack::copy_custom_message(const char* message) {
346  // Invariant: if kErrorCodeOk, no more processing
347  if (LIKELY(error_code_ == kErrorCodeOk)) {
348  return;
349  }
350 
351  clear_custom_message();
352  if (message) {
353  // do NOT use strdup to make sure new/delete everywhere.
354  size_t len = std::strlen(message);
355  char *copied = new char[len + 1]; // +1 for null terminator
356  if (copied) {
357  custom_message_ = copied;
358  std::memcpy(copied, message, len + 1);
359  }
360  }
361 }

363 inline void ErrorStack::append_custom_message(const char* more_custom_message) {
364  // Invariant: if kErrorCodeOk, no more processing
365  if (LIKELY(error_code_ == kErrorCodeOk)) {
366  return;
367  }
368  // augment custom error message
369  if (custom_message_) {
370  // concat
371  size_t cur_len = std::strlen(custom_message_);
372  size_t more_len = std::strlen(more_custom_message);
373  char *copied = new char[cur_len + more_len + 1];
374  if (copied) {
375  custom_message_ = copied;
376  std::memcpy(copied, custom_message_, cur_len);
377  std::memcpy(copied + cur_len, more_custom_message, more_len + 1);
378  }
379  } else {
380  copy_custom_message(more_custom_message); // just put the new message
381  }
382 }
