73 template <typename T, uint64_t ALIGNMENT>
74 inline T align(T value) {
75  uint64_t left = (value + ALIGNMENT - 1);
76  uint64_t right = -ALIGNMENT;
77  uint64_t result = left & right;
78  ASSERT_ND(result >= static_cast<uint64_t>(value));
79  ASSERT_ND(result % ALIGNMENT == 0);
80  ASSERT_ND(result < static_cast<uint64_t>(value) + ALIGNMENT);
81  return static_cast<T>(result);
82 }
